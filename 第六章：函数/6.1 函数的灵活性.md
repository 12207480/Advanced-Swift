        let animal = ["fish", "dog", "elephant"]
        ///反向比较字符串的大小  我们可以嵌套任意的比较函数  让排序功能更强大！
        let okAnimal = animal.sorted { (lhs, rhs) -> Bool in
            let l = lhs.reversed()
            let r = rhs.reversed()
                ///按顺序比较两个字符串的大小   abc > abb
            return l.lexicographicallyPrecedes(r)
        }
        print(okAnimal)
